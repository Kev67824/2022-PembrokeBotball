void moveServo(int servo, int startPosition, int finalPosition){
  set_servo_position(servo, startPosition);
  enable_servos();
  if (startPosition < finalPosition){
  while (startPosition < finalPosition){
    startPosition++;
    set_servo_position(servo, startPosition);
    msleep(5);
  }
  ao();
  }
  else{
    while (startPosition > finalPosition){
    startPosition--;
    set_servo_position(servo, startPosition);
    msleep(5);
  }
  ao();
  }
